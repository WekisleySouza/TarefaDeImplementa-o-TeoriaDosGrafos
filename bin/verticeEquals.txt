public boolean equals(Vertex otherVertex){
        boolean equalsPreviousEdges = (equalsList(this.previousEdgeLabel, otherVertex.getPreviousEdgeLabel()));
        boolean equalsNextEdges = (equalsList(this.nextEdgeLabel, otherVertex.getNextEdgeLabel()));
        return (equalsPreviousEdges && equalsNextEdges && equalsLabel(otherVertex.getLabel()))? true : false;
    }

    private boolean equalsList(List<Integer> list, List<Integer> otherList){
        if(list.size() != otherList.size()){
            return false;
        }
        for(int i = 0; i < otherList.size(); i++){
            if(list.get(i) != otherList.get(i)){
                return false;
            }
        }
        return true;
    }

    private boolean equalsLabel(int otherLabel){
        return (this.label == otherLabel)? true : false;
    }
